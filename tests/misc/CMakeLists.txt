    APPEND_ARGUMENTS
    -poly "${D20_POLY_1}" -primes 2,617,2341309 -exps 1,1,1
    EXPECT_SHA1 ad426daf55ed3ffdf96058602abb25c95c364476
)

    APPEND_ARGUMENTS
    -poly "${D20_POLY_2}" -primes 2,3,7,13,53,541 -exps 2,4,1,1,1,1
    EXPECT_SHA1 dd24813b875f503cf03f783f6ed8ac9d5aa6d902
)

# The hash should be the same as above

    EXPECT_FAIL
)
